import {Component, ComponentFactoryResolver, OnInit, ViewChild, ViewContainerRef} from '@angular/core';
import {AddSessionModalComponent} from '../modal/add-session-modal/add-session-modal.component';
import {MatDialog, MatDialogRef} from '@angular/material';
import {AddAlgoModalComponent} from '../modal/add-algo-modal/add-algo-modal.component';
import {HttpClient} from '@angular/common/http';
import {SMACardComponent} from '../../shared/smoothing/smacard/smacard.component';
import {SmaComponent} from '../../shared/algorithms/sma/sma.component';
import {WmaComponent} from '../../shared/algorithms/wma/wma.component';
import {PcaComponent} from '../../shared/algorithms/pca/pca.component';
import {DftComponent} from '../../shared/algorithms/dft/dft.component';
import {LinechartComponent} from '../../shared/visualisation/linechart/linechart.component';
import {SessionSpark} from '../../models/SessionSpark';
import {InputAlgo} from '../../models/InputAlgo';
import {Ng4LoadingSpinnerService} from 'ng4-loading-spinner';
import {Algo} from '../../models/Algo';

@Component({
  selector: 'app-algorithm',
  templateUrl: './algorithm.component.html',
  styleUrls: ['./algorithm.component.scss']
})
export class AlgorithmComponent implements OnInit {
  matDialogRef: MatDialogRef<AddAlgoModalComponent>;
  selectedAlgo: string;
  sparkSession: number;
  pysparksession: number;
  url_ss = 'http://127.0.0.1:5000/';
  url_db =  'http://127.0.0.1:5001/';
  url_alg = 'http://127.0.0.1:5002/';
  user: string;
  inputs = [];
  constructor(private resolver: ComponentFactoryResolver, private http: HttpClient, private matDialog: MatDialog, private spinnerService: Ng4LoadingSpinnerService) { }
  @ViewChild('container' , {static: true, read: ViewContainerRef}) container;

  ngOnInit() {
    this.user = sessionStorage.getItem('username');
    this.spinnerService.show();
    this.getsession();
    this.getInput();
    console.log(this.inputs);
    this.loadOldAlgo();
  }
  OpenModal() {
    this.matDialogRef = this.matDialog.open(AddAlgoModalComponent, {
      disableClose: true,
      data: {algo : this.selectedAlgo}
    });

    this.matDialogRef.afterClosed().subscribe(res => {
      console.log(res);
      if (res) {
        this.createCardAlgo(res);
      }
    });
  }
  createCardAlgo(algo) {
    switch (algo.toLowerCase()) {
      case 'sma' :
        this.addSma();
        break;
      case 'wma':
        this.addWma();
        break;
      case 'pca':
        this.addPca();
        break;
      case 'dft':
        this.addDft();
        break;
      case 'linechart':
        this.addLineChart();
        break;
    }
  }
  addSma() {
    const ref = this.container.createComponent(this.resolver.resolveComponentFactory(SmaComponent));
    ref.instance.ref = ref;
    ref.instance.inputs = this.inputs
  }
  addWma() {
    const ref = this.container.createComponent(this.resolver.resolveComponentFactory(WmaComponent));
    ref.instance.ref = ref;
  }
  addPca() {
    const ref = this.container.createComponent(this.resolver.resolveComponentFactory(PcaComponent));
    ref.instance.ref = ref;
    ref.instance.inputs = this.inputs;
    console.log(this.sparkSession);
    ref.instance.session_id = this.sparkSession;

  }
  addDft() {
    const ref = this.container.createComponent(this.resolver.resolveComponentFactory(DftComponent));
    ref.instance.ref = ref;
  }
  addLineChart() {
    const ref = this.container.createComponent(this.resolver.resolveComponentFactory(LinechartComponent));
    ref.instance.ref = ref;
  }
  private getsession() {
    this.http.get(this.url_ss + 'getSessions/' + this.user).subscribe((res: SessionSpark[]) => {
      // console.log(res);
      res.forEach(e => {
        if (e.kind == 'pyspark' && e.state == 'idle') {
          this.pysparksession = e.id;
        }
        if (e.kind == 'spark' && e.state == 'idle') {
          this.sparkSession = e.id;
        }
      });
    });
  }
  private getInput() {
    this.getAlgOutPut();
    this.getDatabase();
  }
  private getDatabase() {
    this.http.get(this.url_db + 'getTablesParam/' + this.user).subscribe((res: InputAlgo[] ) => {
      res.forEach(e => {
        this.inputs.push(e);
        console.log(e);
      });
      this.spinnerService.hide();
    });
  }
  private getAlgOutPut() {
    this.http.get(this.url_alg + this.user + '/getOut').subscribe((res: InputAlgo[] ) => {
      res.forEach(e => {
        this.inputs.push(e);

      });
    });
  }

  private loadOldAlgo() {
    this.http.get(this.url_alg + this.user + '/getAll').subscribe((res: Algo[]) => {
      res.forEach(e => {
        switch (e.name.toLowerCase()) {
          case 'sma' :
            const refSMA = this.container.createComponent(this.resolver.resolveComponentFactory(SmaComponent));
            refSMA.instance.ref = refSMA;
            refSMA.instance.inputs = this.inputs;
            refSMA.instance.output = e.output;
            refSMA.instance.idAlgo = e.id ;
            refSMA.instance.isCreated = true;
            break;
          case 'wma':
            this.addWma();
            break;
          case 'pca':
            const refPCA = this.container.createComponent(this.resolver.resolveComponentFactory(SmaComponent));
            refPCA.instance.ref = refPCA;
            refPCA.instance.inputs = this.inputs;
            refPCA.instance.output = e.output;
            refPCA.instance.idAlgo = e.id ;
            refPCA.instance.isCreated = true ;

            break;
          case 'dft':
            this.addDft();
            break;
          case 'linechart':
            this.addLineChart();
            break;
        }
      });
    });
  }
}
