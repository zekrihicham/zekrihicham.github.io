  private addData() {
    for (let i = 0 ; i < 2 ; i++) {
      this.chartOptions.series.push({
        color : this.colors[i],
        name : this.names[i],
        data : this.datas[i]
      });
    }
  }
  chunk(arr, len) {
    let chunks = [],
      i = 0,
      n = arr.length;

    while (i < n) {
      chunks.push(arr.slice(i, i += len));
    }
    // tslint:disable-next-line:triple-equals
    if (chunks[chunks.length - 1].length != chunks[chunks.length - 2].length) {
      chunks[chunks.length - 2] = chunks[chunks.length - 2].concat(chunks[chunks.length - 1] );
      if (chunks.length - 1 > -1) {
        chunks.splice(chunks.length - 1, 1);
      }
    }
    return chunks;
  }

  drop() {
    this.ref.destroy();
  }
