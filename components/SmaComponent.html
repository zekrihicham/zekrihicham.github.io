import {AfterContentInit, Component, ComponentFactoryResolver, Input, OnInit, ViewChild, ViewContainerRef} from '@angular/core';
import {LinechartComponent} from '../../visualisation/linechart/linechart.component';
import {ScatterchartComponent} from '../../visualisation/scatterchart/scatterchart.component';
import {htmlAstToRender3Ast} from '@angular/compiler/src/render3/r3_template_transform';
import {HttpClient} from '@angular/common/http';
import {InputAlgo} from '../../../models/InputAlgo';
import {PcaComponent} from '../pca/pca.component';

@Component({
  selector: 'app-sma',
  templateUrl: './sma.component.html',
  styleUrls: ['./sma.component.scss']
})
export class SmaComponent implements OnInit, AfterContentInit {
  @Input() ref: any;
  @ViewChild('container' , {static: false, read: ViewContainerRef}) container;
  @Input() inputs: InputAlgo[];
  @Input()idAlgo: any;
  outputrep: any;
  @Input() output: any;
  url_ss = 'http://127.0.0.1:5000/';
  url_db =  'http://127.0.0.1:5001/';
  url_alg = 'http://127.0.0.1:5002/';
  url = 'http://127.0.0.1:5003/';
  user: string;
  selectedTable: string;
  winsize: any;
  btnRun = true;
  dropdownList: any;
  selectedItems: any;
  dropdownSettings: any;
  params = '';
  inputtype: string;
  @Input()isCreated: boolean;

  constructor(private http: HttpClient, private resolver: ComponentFactoryResolver) { }

  ngOnInit() {
    this.user = sessionStorage.getItem('username');
    this.createAlgo();

    this.dropdownSettings = {
      singleSelection: false,
      idField: 'item_id',
      textField: 'item_text',
      selectAllText: 'Select All',
      unSelectAllText: 'UnSelect All',
      itemsShowLimit: 3,
      allowSearchFilter: true
    };

  }

  deleteCard() {
    this.http.get(this.url_alg + this.user + '/delete/' + this.idAlgo).subscribe(res => {
      console.log(res);
    });
    this.ref.destroy();
  }

  ngAfterContentInit(): void {
  }
  createAlgo(){
    if (!this.isCreated) {
      this.http.get(this.url_alg + this.user + '/create/sma').subscribe(res => {
        this.idAlgo = res ;
        this.isCreated = true;
      });
    } else  {
      this.outputrep = '';
      this.getoutputrep(this.output);
    }

  }

  changetable() {
    this.dropdownList = [];
    this.inputs.forEach(e => {
      if (e.name == this.selectedTable ) {
        e.params.forEach( p => {
          this.dropdownList.push({ item_id: p, item_text: p })
        });
      }
    });
  }

  run() {
    this.params = '';
    this.selectedItems.forEach(e=>{
      this.params += e.item_id + ',';
    });
    this.inputs.forEach(e => {
      if (e.name == this.selectedTable) {
        this.inputtype = e.typei;
      }
    });
    const body = {inputtype: this.inputtype, params:this.params,tablename:this.selectedTable,winsize:this.winsize};
    console.log(body);
    const formData = new FormData();
    formData.append('inputtype' ,this.inputtype);
    formData.append('params' ,this.params);
    formData.append('tablename' ,this.selectedTable);
    formData.append('winsize' ,this.winsize);
    this.btnRun = false;
    this.http.post(this.url + this.user + '/' + this.idAlgo,formData).subscribe(res => {
      this.outputrep = '';
      this.getoutputrep(res);
      this.output = res;
      this.btnRun = true;
    });


  }
  getoutputrep(res) {
    res.forEach(e => {
      this.outputrep += 'Array (' + e.length + ') , '
    });
  }

  visualiserLine() {
    const ref = this.container.createComponent(this.resolver.resolveComponentFactory(LinechartComponent));
    ref.instance.ref = ref;
    ref.instance.data = this.output;
    ref.instance.names = ['sma01','sma02'];
  }
}
