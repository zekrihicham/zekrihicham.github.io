import {Component, ComponentFactoryResolver, Input, OnInit, ViewChild, ViewContainerRef} from '@angular/core';
import {LinechartComponent} from '../../visualisation/linechart/linechart.component';
import {HttpClient} from '@angular/common/http';
import {InputAlgo} from '../../../models/InputAlgo';
import {ScatterchartComponent} from '../../visualisation/scatterchart/scatterchart.component';

@Component({
  selector: 'app-pca',
  templateUrl: './pca.component.html',
  styleUrls: ['./pca.component.scss']
})
export class PcaComponent implements OnInit {
  url_ss = 'http://127.0.0.1:5000/';
  url_db =  'http://127.0.0.1:5001/';
  url_alg = 'http://127.0.0.1:5002/';
  url = 'http://127.0.0.1:5004/';
  user: string;
  @ViewChild('container' , {static: false, read: ViewContainerRef}) container;
  @Input() inputs: InputAlgo[];
  @Input() ref: any;
  @Input() output: any;
  outputrep: string;
  selectedTabe: string;
  dropdownList: any;
  selectedItems: any;
  dropdownSettings: any;
  params = '';
  inputtype: string;
  reductionSize: any;
  @Input() idAlgo: any;
  btnRun = true;
  @Input() isCreated:boolean;
  @Input() session_id: any;
  selectedTarget: any;
  taget: any;
  constructor(private http: HttpClient, private resolver: ComponentFactoryResolver) { }

  ngOnInit() {
    this.user = sessionStorage.getItem('username');
    // this.createAlgo();

    this.dropdownSettings = {
      singleSelection: false,
      idField: 'item_id',
      textField: 'item_text',
      selectAllText: 'Select All',
      unSelectAllText: 'UnSelect All',
      itemsShowLimit: 3,
      allowSearchFilter: true
    };
  }

  deleteCard() {
    this.http.get(this.url_alg + this.user + '/delete/' + this.idAlgo).subscribe(res => {
      console.log(res);
    });
    this.ref.destroy();  }

  createAlgo(){
    if (!this.isCreated) {
      this.http.get(this.url_alg + this.user + '/create/pca').subscribe(res => {
        this.idAlgo = res ;
        this.isCreated = true;
      });
    } else  {
      this.outputrep = '';
      this.getoutputrep(this.output);
    }
  }

  changetable() {
    this.dropdownList = [];
    this.inputs.forEach(e => {
      if (e.name == this.selectedTabe ) {
        e.params.forEach( p => {
          this.dropdownList.push({ item_id: p, item_text: p })
        });
      }
    });
  }

  run() {
    this.params = '';
    this.selectedItems.forEach(e=>{
      this.params += e.item_id + ',';
    });
    this.inputs.forEach(e => {
      if (e.name == this.selectedTabe) {
        this.inputtype = e.typei;
      }
    });
    const formData = new FormData();
    formData.append('inputtype' ,this.inputtype);
    formData.append('params' ,this.params);
    formData.append('tablename' ,this.selectedTabe);
    formData.append('reductionsize' , this.reductionSize);
    formData.append('session_id' , this.session_id);

    this.btnRun = false;
    this.http.post(this.url + this.user + '/' + this.idAlgo,formData).subscribe(res => {
      this.outputrep = '';
      this.getoutputrep(res);
      this.output = res;
      console.log(this.output);
      this.btnRun = true;
    });
  }
  getoutputrep(res) {

    this.outputrep += 'Array (' + res.length + ') , '

  }

  visualiserScatter() {
    const ref = this.container.createComponent(this.resolver.resolveComponentFactory(ScatterchartComponent));
    ref.instance.ref = ref;
    ref.instance.data = [[1,2],[4,55],[22,5],[12,10]]
    ref.instance.targets =['1','2','3','4'];
  }

  changetarget() {
    this.http.get(this.url + this.user + '/' + this.inputtype + '/' + this.selectedTabe + '/' + this.selectedTarget ).subscribe(res => {
      this.taget = res;
    });
  }
}
